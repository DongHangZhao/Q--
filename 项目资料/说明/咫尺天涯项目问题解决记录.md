# 咫尺天涯项目问题解决记录

## 1. 数据库兼容性问题

### 1.1 问题：时间戳格式不一致导致的TypeError
- **问题描述**: 
  - 启动应用时出现 `TypeError: fromisoformat: argument must be str` 错误
  - 原因是数据库中存在多种时间戳格式（字符串、数字、标准datetime等）
  - SQLAlchemy无法统一解析不同格式的时间戳

- **解决方案**:
  - 创建自定义FlexibleDateTime类型装饰器
  - 实现process_result_value方法处理多种格式
  - 添加对字符串、数字、标准datetime格式的支持

- **代码实现**:
```python
class FlexibleDateTime(TypeDecorator):
    """
    自定义灵活的日期时间类型，能处理多种时间戳格式
    """
    impl = DateTime
    cache_ok = True

    def process_result_value(self, value, dialect):
        if value is None or value == '':
            return datetime.utcnow()
        if isinstance(value, str):
            # 处理不同格式的时间戳字符串
            if '.' in value:
                # 处理包含毫秒的标准格式: 'YYYY-MM-DD HH:MM:SS.ffffff'
                parts = value.split('.')
                dt_part = parts[0]
                microsec_part = parts[1][:6] if len(parts) > 1 else '0'  # 限制微秒为6位
                
                # 解析日期时间部分
                dt = datetime.strptime(dt_part, '%Y-%m-%d %H:%M:%S')
                # 添加微秒
                return dt.replace(microsecond=int(microsec_part.ljust(6, '0')))
            else:
                # 解析不含微秒的格式
                return datetime.strptime(value, '%Y-%m-%d %H:%M:%S')
        elif isinstance(value, (int, float)):
            # 如果是数字时间戳，转换为datetime
            try:
                return datetime.fromtimestamp(value)
            except (ValueError, OSError):
                return datetime.utcnow()
        return value
```

- **效果**: 成功解决了数据库时间戳格式兼容性问题，应用可以正常启动

# 数据库合并脚本的关键部分
def merge_databases(source_db, target_db):
    # 连接源数据库和目标数据库
    source_conn = sqlite3.connect(source_db)
    target_conn = sqlite3.connect(target_db)
    
    # 获取所有表名
    tables = ['users', 'posts', 'videos', 'messages', 'news', 'comments', 'follows']
    
    for table in tables:
        # 从源数据库读取数据
        df_source = pd.read_sql_query(f"SELECT * FROM {table}", source_conn)
        df_target = pd.read_sql_query(f"SELECT * FROM {table}", target_conn)
        
        # 合并数据（避免重复）
        merged_df = pd.concat([df_target, df_source]).drop_duplicates(subset=['id'], keep='first')
        
        # 写入目标数据库
        merged_df.to_sql(table, target_conn, if_exists='replace', index=False)
    
    source_conn.close()
    target_conn.close()
```

- **效果**: 成功保留了原始用户数据，同时合并了新功能，用户数据完整性得到保障

def allowed_file(filename):
    """检查文件扩展名是否允许"""
    ALLOWED_EXTENSIONS = {'png', 'jpg', 'jpeg', 'gif', 'mp4', 'avi', 'mov', 'pdf', 'doc', 'docx'}
    return '.' in filename and \
           filename.rsplit('.', 1)[1].lower() in ALLOWED_EXTENSIONS

def secure_filename(filename):
    """生成安全的文件名"""
    # 移除路径字符，只保留文件名
    filename = os.path.basename(filename)
    # 替换非法字符
    filename = re.sub(r'[^\w\s.-]', '', filename).strip()
    # 生成唯一文件名
    name, ext = os.path.splitext(filename)
    unique_name = f"{name}_{uuid.uuid4().hex[:8]}{ext}"
    return unique_name
```

- **效果**: 文件上传功能更加安全，有效防范了安全风险

- **代码实现**:
```python
from threading import Lock

like_lock = Lock()

def toggle_like(user_id, content_id, content_type):
    """线程安全的点赞操作"""
    with like_lock:
        if content_type == 'post':
            existing_like = PostLike.query.filter_by(user_id=user_id, post_id=content_id).first()
            content = Post.query.get(content_id)
        # 类似的其他类型处理
        
        if existing_like:
            db.session.delete(existing_like)
            content.likes_count -= 1
        else:
            # 添加新的点赞
            if content_type == 'post':
                like = PostLike(user_id=user_id, post_id=content_id)
            # 其他类型处理
            db.session.add(like)
            content.likes_count += 1
        
        db.session.commit()
```
