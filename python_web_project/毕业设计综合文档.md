class NewsScraperV2:
    """优化版新闻抓取器"""
    
    def __init__(self):
        self.sources = [
            {'name': '新华网', 'url': 'http://www.news.cn/'},
            {'name': '央视新闻', 'url': 'https://news.cctv.com/'},
            {'name': '澎湃新闻', 'url': 'https://www.thepaper.cn/'}
        ]
    
    def scrape_from_source(self, source):
        """从指定来源抓取新闻"""
        response = requests.get(source['url'], timeout=10)
        soup = BeautifulSoup(response.text, 'html.parser')
        
        news_items = []
        for article in soup.find_all('a', href=True):
            if self._is_valid_news_link(article):
                news = {
                    'title': article.get_text(strip=True),
                    'url': article['href'],
                    'source': source['name'],
                    'timestamp': datetime.now()
                }
                news_items.append(news)
        
        return news_items
```

@news_bp.route('/search')
def news_search():
    """新闻搜索（支持关键字和日期）"""
    keyword = request.args.get('keyword', '')
    search_date = request.args.get('date', '')
    source = request.args.get('source', '')
    
    query = News.query
    
    # 关键字搜索
    if keyword:
        query = query.filter(
            db.or_(
                News.title.ilike(f'%{keyword}%'),
                News.content.ilike(f'%{keyword}%'),
                News.summary.ilike(f'%{keyword}%')
            )
        )
    
    # 日期筛选
    if search_date:
        target_date = datetime.strptime(search_date, '%Y-%m-%d').date()
        query = query.filter(db.func.date(News.timestamp) == target_date)
    
    # 来源筛选
    if source:
        query = query.filter(News.source == source)
    
    news_items = query.order_by(News.timestamp.desc()).paginate(
        page=request.args.get('page', 1, type=int), 
        per_page=10, 
        error_out=False
    )
    
    return render_template('news_search.html', 
                         news_items=news_items.items,
                         pagination=news_items,
                         keyword=keyword,
                         search_date=search_date,
                         selected_source=source)

@app.route('/like_news/<int:news_id>', methods=['POST'])
@login_required
def like_news(news_id):
    """点赞新闻"""
    news = News.query.get_or_404(news_id)
    
    old_likes_count = news.likes_count or 0
    
    # 检查是否已点赞
    existing_like = NewsLike.query.filter_by(
        user_id=current_user.id, news_id=news_id).first()
    
    if existing_like:
        # 取消点赞
        db.session.delete(existing_like)
        news.likes_count = max(0, news.likes_count - 1)
        liked = False
        operation_type = 'unlike'
    else:
        # 添加点赞
        like = NewsLike(user_id=current_user.id, news_id=news_id)
        db.session.add(like)
        news.likes_count += 1
        liked = True
        operation_type = 'like'
    
    db.session.commit()
    
    # 记录操作日志
    log = NewsOperationLog(
        news_id=news.id,
        user_id=current_user.id,
        operation_type=operation_type,
        old_value=old_likes_count,
        new_value=news.likes_count,
        ip_address=request.remote_addr,
        user_agent=request.headers.get('User-Agent')
    )
    db.session.add(log)
    db.session.commit()
    
    return jsonify({'liked': liked, 'likes_count': news.likes_count})
```

#### 热度计算算法
```python
def calculate_trend_score(likes_count, views, comments_count, age_hours):
    """
    计算热度分数
    基于点赞数、浏览数、评论数和内容年龄
    """
    # 基础分数计算（加权）
    base_score = (likes_count * 3) + (views * 0.5) + (comments_count * 2)

    # 时间衰减因子
    decay_factor = max(0.1, 24.0 / (age_hours + 1))

    # 最终得分
    final_score = base_score * decay_factor
    return round(final_score, 2)
```

**算法说明：**
- 点赞权重：3 分
- 浏览权重：0.5 分
- 评论权重：2 分
- 时间衰减：越新的内容权重越高

@news_bp.route('/calendar')
def news_calendar():
    """新闻日历视图"""
    year = request.args.get('year', datetime.now().year, type=int)
    month = request.args.get('month', datetime.now().month, type=int)
    
    # 获取当月新闻
    first_day = date_obj(year, month, 1)
    last_day = date_obj(year, month + 1, 1) - timedelta(days=1) if month != 12 \
               else date_obj(year + 1, 1, 1) - timedelta(days=1)
    
    monthly_news = News.query.filter(
        News.timestamp >= first_day,
        News.timestamp <= last_day
    ).all()
    
    # 按日期分组
    news_by_date = {}
    for news in monthly_news:
        date_key = news.timestamp.date()
        if date_key not in news_by_date:
            news_by_date[date_key] = []
        news_by_date[date_key].append(news)
    
    cal = calendar.monthcalendar(year, month)
    
    return render_template('news_calendar.html',
                         year=year, month=month,
                         calendar_days=cal,
                         news_by_date=news_by_date)

def migrate():
    """执行数据库迁移"""
    with app.app_context():
        # 1. 添加 comments_count 列
        db.session.execute(text(
            "ALTER TABLE news ADD COLUMN comments_count INTEGER DEFAULT 0"
        ))
        db.session.commit()
        
        # 2. 创建操作日志表
        create_table_sql = """
        CREATE TABLE IF NOT EXISTS news_operation_logs (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            news_id INTEGER NOT NULL,
            user_id INTEGER,
            operation_type VARCHAR(20) NOT NULL,
            old_value INTEGER,
            new_value INTEGER,
            timestamp DATETIME DEFAULT CURRENT_TIMESTAMP,
            ip_address VARCHAR(50),
            user_agent VARCHAR(200),
            FOREIGN KEY (news_id) REFERENCES news(id),
            FOREIGN KEY (user_id) REFERENCES users(id)
        )
        """
        db.session.execute(text(create_table_sql))
        db.session.commit()
        
        # 3. 初始化现有数据
        db.session.execute(text("""
            UPDATE news 
            SET comments_count = (
                SELECT COUNT(*) FROM comments 
                WHERE comments.news_id = news.id
            )
        """))
        db.session.commit()

def news_scraper_job():

```sql
-- 查看新闻统计

WHERE op.news_id = 18
ORDER BY op.timestamp ASC;

-- 统计今日操作分布
SELECT 
    operation_type,
    COUNT(*) as count
FROM news_operation_logs
WHERE DATE(timestamp) = DATE('now')
GROUP BY operation_type;
```

        news_list = self.scrape_from_source(source)
        for news_data in news_list:

            existing = News.query.filter_by(
